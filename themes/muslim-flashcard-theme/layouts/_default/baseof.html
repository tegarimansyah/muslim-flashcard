// Doa viewer functions function showDoaModal(groupId, cardId) { if (!window.doaData || !window.doaData.groups) { console.error('Doa data not loaded'); return; } const groupData = window.doaData.groups.find(g => g.id === groupId); if (!groupData) { console.error('Group not found:', groupId); return; } currentGroup = groupId; currentDoaData = groupData; // Find card index currentCardIndex = groupData.cards.findIndex(card => card.id === cardId); if (currentCardIndex === -1) { currentCardIndex = 0; } // Update group info in header document.getElementById('viewer-group-icon').style.backgroundColor = groupData.color; document.getElementById('viewer-group-icon').textContent = groupData.icon; document.getElementById('viewer-group-title').textContent = groupData.title; // Build doa slides const slideContainer = document.getElementById('doa-slide-container'); slideContainer.innerHTML = ''; groupData.cards.forEach(card => { const slide = document.createElement('div'); slide.className = 'doa-slide'; slide.innerHTML = `

${card.title}

${card.latin}

${card.arabic}
`; slideContainer.appendChild(slide); }); // Build navigation dots const dotsContainer = document.querySelector('.navigation-dots'); dotsContainer.innerHTML = ''; groupData.cards.forEach((card, index) => { const dot = document.createElement('div'); dot.className = 'nav-dot' + (index === currentCardIndex ? ' active' : ''); dot.onclick = () => goToSlide(index); dotsContainer.appendChild(dot); }); // Show initial slide goToSlide(currentCardIndex); openModal('doa-viewer-modal'); }
